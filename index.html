    <script>

        // 喝酒懲罰清單

        const drinkRules = [

            "自罰一杯",

            "跟左邊的人碰杯一起喝",

            "跟右邊的人碰杯一起喝",

            "指定一位朋友陪你喝",

            "全場一起喝",

            "男生全喝",

            "女生全喝",

            "玩手機的人喝一杯",

            "最後一個到的人喝",

            "笑最大聲的人喝",

            "連喝兩杯",

            "隨機找一人猜拳，輸的喝",

            "講一個笑話，不好笑就喝"

        ];


        // 隨機抽人

        function randomPerson() {

            const playerList = getPlayerList();

            if (playerList.length === 0) {

                showResult("⚠️ 請先輸入玩家名字！");

                return;

            }

            const lucky = playerList[Math.floor(Math.random() * playerList.length)];

            showResult(`🍺 幸運得主：${lucky}`, "趕快準備喝酒～");

        }


        // 隨機喝酒指令

        function randomDrink() {

            const rule = drinkRules[Math.floor(Math.random() * drinkRules.length)];

            showResult(`🥳 指令：${rule}`);

        }


        // 隨機幾杯

        function randomCount() {

            const count = Math.floor(Math.random() * 5) + 1;

            showResult(`🍷 喝 ${count} 杯！`, "乾啦～");

        }


        // 取得玩家清單

        function getPlayerList() {

            const input = document.getElementById("players").value.trim();

            return input ? input.split(",").map(name => name.trim()) : [];

        }


        // 顯示結果

        function showResult(text, tip = "") {

            document.getElementById("result").innerText = text;

            document.getElementById("tip").innerText = tip;

        }


        // 重置

        function reset() {

            showResult("點按按鈕開始遊戲", "");

        }
